  \begin{proof}

\section{RÃĐsolution de la SDE : trois variantes}

\subsection{Variance Exploding (VE)}

  \begin{proof}

\subsection{Variance Preserving (VP) --- Ornstein--Uhlenbeck}

\end{hypothesebox}

\textbf{Ãtape 5 : identification et vÃĐrification.} Par la Proposition \ref{prop:gauss} : $X_t \mid X_0 \sim \mathcal{N}(e^{-t} X_0,\; (1-e^{-2t})\, I_n)$. \end{proof}


\subsection{DDPM (Denoising Diffusion Probabilistic Models)--- formulation discrÃĻte}

\begin{propositionbox}[Forme du score de mesure]

\end{proof}

\begin{proof}

\subsubsection{Distribution de rÃĐfÃĐrence pour $x_{0,i}$}

Le calcul de $q_i^r(x_{0,i})$ est plus dÃĐlicat car il fait intervenir des formes linÃĐaires et quadratiques. Nous utilisons ici les techniques de calculs de Zheng-Rodet \cite{rodet} qui permet de traiter systÃĐmatiquement ces termes.

\begin{resultbox}[Technique 1 : Zheng-Rodet]
\label{prop:linear_integral}
Pour tout vecteur $v \in \mathbb{R}^n$ :
\begin{equation}
    \int x^T v \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j = x_i\, v_i + C_1
    \label{eq:linear_trick}
\end{equation}
oÃđ $C_1 = \sum_{j \neq i} \mu_j^k\, v_j$ est une constante indÃĐpendante de $x_i$.
\end{resultbox}

\begin{proof}

On dÃĐveloppe le produit scalaire :
\[
x^T v = \sum_{j=1}^n x_j\, v_j = x_i\, v_i + \sum_{j \neq i} x_j\, v_j
\]
En intÃĐgrant sur les $x_j$ pour $j \neq i$ :
\[
\int x^T v \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j = x_i\, v_i + \sum_{j \neq i} v_j \underbrace{\int x_j\, q_j^k(x_j)\, \mathrm{d}x_j}_{= \mu_j^k} = x_i\, v_i + \sum_{j \neq i} \mu_j^k\, v_j
\]
\end{proof}

\begin{resultbox}[Technique 2 : Zheng-Rodet]
\label{prop:quadratic_integral}
Pour toute matrice symÃĐtrique $H \in \mathbb{R}^{n \times n}$ :
\begin{equation}
    \boxed{\int x^T H x \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j = x_i^2\, H_{ii} + 2x_i\, (H\boldsymbol{\mu}^k)_i - 2x_i\, H_{ii}\, \mu_i^k + C_2}
    \label{eq:quadratic_trick}
\end{equation}
oÃđ $C_2$ est une constante indÃĐpendante de $x_i$ et $\boldsymbol{\mu}^k = (\mu_1^k, \ldots, \mu_n^k)^T$.
\end{resultbox}

\begin{proof}

\textbf{Ãtape 1 : dÃĐveloppement de la forme quadratique.}
\[
x^T H x = \sum_{p=1}^n \sum_{l=1}^n H_{pl}\, x_p\, x_l
\]

\textbf{Ãtape 2 : intÃĐgration sur les $x_j$ pour $j \neq i$.} On sÃĐpare les termes selon qu'ils dÃĐpendent de $x_i$ ou non :
\begin{align}
    \int x^T H x \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j &= \int \left[x_i^2 H_{ii} + 2x_i \sum_{p \neq i} H_{ip} x_p + \sum_{\substack{p,l \\ p \neq i, l \neq i}} H_{pl} x_p x_l\right] \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j
\end{align}

\textbf{Ãtape 3 : ÃĐvaluation terme par terme.}

\textit{Premier terme :} $x_i^2 H_{ii}$ ne dÃĐpend pas des $x_j$ pour $j \neq i$, donc :
\[
\int x_i^2 H_{ii} \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j = x_i^2 H_{ii}
\]

\textit{DeuxiÃĻme terme :} Par sÃĐparabilitÃĐ de $q^k$ :
\[
\int 2x_i \sum_{p \neq i} H_{ip} x_p \prod_{j \neq i} q_j^k(x_j)\, \mathrm{d}x_j = 2x_i \sum_{p \neq i} H_{ip} \mu_p^k
\]

\textit{TroisiÃĻme terme :} Il ne dÃĐpend pas de $x_i$, donc c'est une constante $C_2'$.

\textbf{Ãtape 4 : rÃĐÃĐcriture.} On a :
\[

\]
D'oÃđ :
\[

\]

\end{proof}
